表结构
表名		ENVIRONMENT   项目环境
		N_ID 项目id
		N_PROJECT_ID 项目id
		S_NAME     环境名称
		S_DOMAIN   环境域名

表名 	PROJECT  项目
		N_ID 项目id
		S_NAME 项目名称
		D_CREATE 创建日期

表名		FUNCTION  模块
		N_ID 模块id
		N_PROJECT_ID  项目id
		S_NAME  功能或模块名称
		D_CREATE  创建时间

表名		TESTCASE  测试用例
		N_ID 用例Id
		N_FUNCTION_ID 功能或模块id
		S_NAME  测试名称，说明
		S_DESC 详细说明
		S_STEP 步骤
		S_URL 测试接口
		S_TYPE  接口访问类型(get/post)
		S_DATA  测试参数及数据
		S_EXPECT 期望值
		D_CREATE 创建时间
		S_STATUS  状态(1 启用，0禁用)

DROP TABLE IF EXISTS ENVIRONMENT ;		
CREATE TABLE ENVIRONMENT(N_ID INT IDENTITY PRIMARY KEY,N_PROJECT_ID INT, S_NAME VARCHAR(100),S_DOMAIN VARCHAR(100));


DROP TABLE IF EXISTS PROJECT ;	
CREATE TABLE PROJECT(N_ID INT IDENTITY PRIMARY KEY,S_NAME VARCHAR(200),D_CREATE DATE);

		
DROP TABLE IF EXISTS FUNCTION;		
CREATE TABLE FUNCTION(N_ID INT IDENTITY PRIMARY KEY,N_PROJECT_ID INT,S_NAME VARCHAR(200),D_CREATE DATE);
		
DROP TABLE IF EXISTS TESTCASE;		
CREATE TABLE TESTCASE(N_ID INT IDENTITY PRIMARY KEY
					,N_FUNCTION_ID INT
					,S_NAME VARCHAR(200)
					,S_DESC VARCHAR(2000)
					,S_STEP VARCHAR(4)
					,S_URL VARCHAR(500)
					,S_TYPE VARCHAR(10)
					,S_DATA VARCHAR(2000)
					,S_EXPECT VARCHAR(1000)
					,D_CREATE TIMESTAMP
					,S_STATUS VARCHAR(2));	
					
ALTER TABLE TESTCASE ALTER COLUMN D_CREATE TIMESTAMP;
